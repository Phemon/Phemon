【题解】病毒入侵

· 2019-05-06 · # OI # 题解

这数据强度，每次询问跑一遍肯定是不行的，考虑把询问离线。

显然，可以把询问和给出的边都按照v从小到大排序，依次考虑每个询问，对于每次询问，加入不大于该询问的v的边，同时使用一个并查集进行统计，同时统计该联通块点个数、最大值和最大值个数。

这种模拟，当时敲了四十多分钟，还是太菜了！ 压着2s时限ac的代码：

#include <bits/stdc++.h>
using namespace std;
const int N = 300010, M = 400000;
int n, m, q;
int a[N], b[N], imp[N], cnt[N], ga[N], v[N];
int p[3000010], fa[N];
struct node{
    int x, y, v;
} l[M];
struct att{
    int n, v, st, i;
} g[100010];
struct Ans{
    int A, C, B;
} ans[100010];
bool cmp(node a, node b){
    return a.v < b.v;
}
bool cmp2(att a, att b){
    return a.v < b.v;
}
int find(int x){
    if (fa[x] == x)
        return x;
    else
        return fa[x] = find(fa[x]);
}
long long sumb;
int main(){
    cin >> n >> m >> q;
    for (int i = 1; i <= n; i++)
        scanf("%d", &a[i]);
    for (int i = 1; i <= n; i++)
        scanf("%d", &b[i]), sumb += b[i];
    for (int i = 1; i <= m; i++)
        scanf("%d%d%d", &l[i].x, &l[i].y, &l[i].v);
    for (int i = 1; i <= N; i++){
        fa[i] = i;
        imp[i] = a[i];
        cnt[i] = 1;
        ga[i] = b[i];
    }
    for (int i = 1; i <= q; i++){
        scanf("%d%d", &g[i].n, &g[i].v);
        g[i].i = i;
        g[i].st = g[i - 1].st + g[i - 1].n;
        for (int j = g[i].st; j < g[i].st + g[i].n; j++)
            scanf("%d", &p[j]);
    }
    sort(l + 1, l + m + 1, cmp);
    sort(g + 1, g + q + 1, cmp2);
    int ls = 1;
    for (int i = 1; i <= q; i++){
        for (; l[ls].v <= g[i].v && ls <= m; ls++){
            int xx = find(l[ls].x), yy = find(l[ls].y);
            if (xx == yy)
                continue;
            fa[yy] = xx;
            if (imp[xx] == imp[yy])
                cnt[xx] += cnt[yy];
            if (imp[xx] < imp[yy])
                imp[xx] = imp[yy], cnt[xx] = cnt[yy];
            ga[xx] += ga[yy];
        }
        for (int j = g[i].st; j < g[i].st + g[i].n; j++){
            int fp = find(p[j]);
            if (v[fp] == i)
                continue;
            v[fp] = i;
            if (ans[g[i].i].A == imp[fp])
                ans[g[i].i].C += cnt[fp];
            if (ans[g[i].i].A < imp[fp])
                ans[g[i].i].A = imp[fp], ans[g[i].i].C = cnt[fp];
            ans[g[i].i].B += ga[fp];
        }
    }
    for (int i = 1; i <= q; i++)
        printf("%lld %d\n", 1ll * ans[i].A * ans[i].C, sumb - ans[i].B);
}
